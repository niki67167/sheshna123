-- גרסת חירום יא מאניק - בלי אנימציות, נטו עבודה
local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")
local PlaceId = 104715542330896

-- מחיקה של כל מה שהיה קודם כדי שלא יתקע
for _, v in pairs(game.CoreGui:GetChildren()) do
    if v.Name == "Manik_Fix" or v.Name == "Manik_Ultra_V12" or v.Name == "Manik_Hunter_V11" then
        v:Destroy()
    end
end

local ScreenGui = Instance.new("ScreenGui", game.CoreGui)
ScreenGui.Name = "Manik_Fix"

local Main = Instance.new("Frame", ScreenGui)
Main.Size = UDim2.new(0, 300, 0, 250)
Main.Position = UDim2.new(0.5, -150, 0.5, -125)
Main.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
Main.Active = true
Main.Draggable = true -- פשוט ובטוח

local Title = Instance.new("TextLabel", Main)
Title.Size = UDim2.new(1, 0, 0, 50)
Title.Text = "MANIK FIX V1"
Title.TextColor3 = Color3.new(1,1,1)
Title.TextSize = 25
Title.BackgroundTransparency = 1

local Status = Instance.new("TextLabel", Main)
Status.Size = UDim2.new(1, 0, 0, 50)
Status.Position = UDim2.new(0, 0, 0.3, 0)
Status.Text = "מוכן לצוד יא מאניק"
Status.TextColor3 = Color3.fromRGB(255, 0, 100)
Status.BackgroundTransparency = 1

local Hunt = Instance.new("TextButton", Main)
Hunt.Size = UDim2.new(0.8, 0, 0, 60)
Hunt.Position = UDim2.new(0.1, 0, 0.6, 0)
Hunt.Text = "תתחיל לצוד"
Hunt.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
Hunt.TextColor3 = Color3.new(1,1,1)

Hunt.MouseButton1Click:Connect(function()
    Hunt.Text = "חופר..."
    local url = "https://games.roblox.com/v1/games/" .. PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
    local s, r = pcall(function() return HttpService:JSONDecode(game:HttpGet(url)) end)
    if s and r.data then
        for _, v in ipairs(r.data) do
            if v.playing >= 2 and v.playing <= 6 then
                Status.Text = "טס לשרת..."
                TeleportService:TeleportToPlaceInstance(PlaceId, v.id)
                return
            end
        end
        Status.Text = "אין שרתים יא מאניק"
    else
        Status.Text = "שגיאת אינטרנט"
    end
    Hunt.Text = "תתחיל לצוד"
end)
